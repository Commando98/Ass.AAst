package javaapplication29;

public class JavaApplication29 {

    public static void main(String[] args) {
        int i, j, res;
        for (i = 1; i < 10000; i++) {
            res = 0;
            for (j = 1; j < i; j++) {
                if (i % j == 0) {
                    res = res + j;

                }

            }
            if (res == i) {
                System.out.println(+i);
            }
        }

    }
}
