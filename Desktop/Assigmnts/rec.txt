public class rectangle {
    double width , hieght ;
 public   rectangle()
    {
        width = 1; 
        hieght = 1 ;
    }
   public rectangle(double w,double h)
    {
      this.width = w;
      this.hieght = h;
        
    }
    public double getArea()
    {
        return width * hieght ;
        
    }
    public double  getPerimtetre()
    {
        return width * hieght * 2;
    
    }

}
